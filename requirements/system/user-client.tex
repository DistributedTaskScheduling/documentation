\subsection{User Client}


\subsubsection{Functional Requirements} 
\begin{itemize}
	\item	[FR01] Logging in to the central server
	\item	[FR02] Logging out from the central server
	\item	[FR05] Queueing new jobs
	\item	[FR06] Querying the state of jobs
	\item	[FR07] Canceling jobs
	\item	[FR08] Receiving the output
	\item	[FR09] Setting job priorities 
	\item	[FR10] Checking the specifications of available containers
	\item	[FR11] Choosing the parameters of the container(preferred OS/environment)
	\item	[FR12] Choosing the running mode: "blocking mode"/ default mode
	\item	[FR13] Choosing the minimal hardware requirements for a job	
	\item	[FR14] Checking the state of the entire server
\end{itemize}

\subsubsection{Non Functional Requirements} 
\begin{itemize}
	\item	[NFR01] Responses from the server must take less 10ms
	\item	[NFR02] Users can't queue more than 5 jobs at the same time
	
\end{itemize}

\subsubsection{Optional Requirements} 
\begin{itemize}
	\item	[OR01] Saving a private image for future usage
	\item	[OR02] Scheduling a job to run at a specific time
	\item	[OR03] Choosing a preferred container to use
	\item	[OR04] Checking the estimated waiting time
	\item	[OR05] Checking the estimated time of arrival
\end{itemize}

\subsubsection{Product Data} 
\begin{itemize}
	\item [PD01] Log in information
	\item [PD02] Configuration files
	\item [PD03] Currently runnning jobs
	\item [PD04] Outpuf of finished jobs
	
\end{itemize}
